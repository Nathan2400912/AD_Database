Common Queries

Find differentially expressed genes for a condition 
SELECT g.gene_symbol, d.log2foldchange, d.padj 
FROM Genes g
JOIN Differential_Expression d USING (gid)
JOIN Conditions c USING (cdid)
WHERE c.name = 'IFN' AND d.padj < 0.05
Find CREs linked to a specific gene 
SELECT c.chromosome, c.start_position, c.end_position
FROM CRE_Gene_Interactions cgi
JOIN Genes g USING (gid)
JOIN Cis_Regulatory_Elements c USING (cid)
WHERE g.gene_symbol = 'TREM2'
